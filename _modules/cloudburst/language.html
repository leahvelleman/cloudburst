[docs]class Level(object):
    """A level of representation for language forms.

    A level wraps a finite state transducer which converts keys into
    strings. The level provides :func:`lookup` and :func:`lookdown` functions
    to apply this transducer in both directions.

    >>> L = Language(...)
    >>> spelling = L.add_child(...)
    >>> spelling.lookup("guapo")

    >>> spelling.lookdown("handsome[+Adj][+Masc][+Sg]")

    A level also provides convenience functions for defining new levels based
    on it. For instance, if you already have a level whose :func:`lookdown`
    function turns keys into written words, you might use it as the basis for a
    new level representing pronunciation.

    >>> pronunciation = spelling.add_child(...)

    >>> pronunciation.lookup("gera")

    Levels created in this way form a tree, and strings can be converted
    between any two levels that belong to the same tree, using conversion
    functions stored in the :attr:`~Level.converters` attribute.

    >>> pronunciation.converters[spelling]("gera")

    Finally, a level can be called with a string argument to produce a Sphinx
    document tree node representing it. The `&` operator can be used as a
    shortcut to combine several levels of representations in a single document
    tree node.

    >>> spelling("guerra")

    >>> spelling("guerra") & pronunciation

    >>> spelling & pronunciation("gera")

    Attributes:
        fst: If this level is a key level, :attr:`fst` is a Pynini acceptor
            representing the possible keys. If it is a derived level,
            :attr:`fst` is a Pynini transducer deriving it from its parent
            level.
        parent: The parent level that this one is defined in terms of; `None`
            if this is the key level of its language.
        children: A list of levels that are defined in terms of this one.
        converters: If `thislevel` and `otherlevel` are levels in the same
            language, ``thislevel.converters[otherlevel]`` is a transducer
            converting from representations at `thislevel` to representations
            at `otherlevel`.
        language: Language this level belongs to.
    """

    def __init__(self,
                 fst: pynini.Fst,
                 parent: 'Level' = None) -> None:
        self.children = []
#        self.__check_fst(fst)
        self.fst = fst
        self.fmt = lambda x: x
        self.converters = {self: fst.copy().project(project_output=True)}
        if parent:
            self.language = parent.language
            self.parent = parent
            self.parent.children.append(self)
            self.__connect_to_parent(self.parent)
        else:
            self.language = self
            self.parent = None
        self.sigma = get_sigma(self.fst.output_symbols())

    def __connect_to_parent(self, parent: 'Level') -> None:
        """" Compose a converter between myself and each node my parent
        connects to. """
        fst = self.fst
        for target in list(parent.converters):
            target.converters[self] = pynini.compose(
                target.converters[parent], fst)
            self.converters[target] = pynini.compose(
                fst.copy().invert(), parent.converters[target])

    def __call__(self, content: str) -> 'Form':
        values = self.lookup(content)
        if not values:
            raise InconsistentForm
        return Form(values=values, levels_to_show=[self])


[docs]    def lookup(self, form: str) -> Set[str]:
        """Convert the specified form from this level to a set of keys.

        Args:
            form: Form to be converted.

        Returns:
            Set containing a single key if the form is unambiguous, or multiple
            keys if it is ambiguous.
        """
        return apply_up(self.language.converters[self], form)

        Returns:
            Set containing a single form if the key is unambiguous, or

        return apply_down(self.language.converters[self], key)



[docs]    def add_child(self,
                  fst: pynini.Fst) -> 'Level':
        """Add another level derived from this one using the specified FST.

        The :meth:`add_child` method automatically sets up converters between
        the newly created level and all other levels in its tree.

        Args:
            fst: A Pynini transducer which derives forms at the child level
                from forms at this one.

        Returns:
            The newly created level.
        """
        return Level(parent=self, fst=fst)



[docs]    def rewrite(self, change, left_environment="", right_environment="",
                direction="ltr", mode="obl", added_symbols=None) -> 'Level':
        #pylint: disable=too-many-arguments
        """ Create a level derived from this one by a context-dependent
        rewrite rule.

        This method wraps the built-in Pynini :meth:`pynini.cdrewrite`
        method. It provides friendlier argument names (in particular, avoiding
        the keyword `lambda`), takes care of specifying the correct
        `sigma_star`, and ensures that the result is a :class:`Level` object
        instead of a bare Fst.

        Args:
            change: A pynini transducer representing the change to
                be made.
            left_environment: A pynini acceptor or string
                representing the left environment for the change.
            right_environment: A pynini acceptor or
                string representing the right environment for the change.

        Returns:
            The newly created level.
        """
        if not added_symbols:
            added_symbols = []
        new_lower_sigma = u(self.sigma, m(added_symbols))
        return self.add_child(
            pynini.cdrewrite(
                change,
                left_environment,
                right_environment,
                sigma_star=new_lower_sigma.star,
                direction=direction, mode=mode))



[docs]    def walk(self) -> Generator['Level', None, None]:
        """Breadth-first traversal of this level and its children.

        Yields:
            The next level in the breadth-first traversal.
        """
        yield self
        for child in self.children:
            for node in child.walk():
                yield node



[docs]    def walk_with_depth(self, level: int = 0) -> Generator[Tuple['Level', int],
                                                           None, None]:
        """Breadth-first traversal of this level and its children, keeping
        track of the depth within the tree. Used to pretty-print the levels
        that make up a Language.

        Yields:
            The next level in the breadth-first traversal, and a number
            indicating its depth, yielded as a tuple.
        """
        yield (self, level)
        for child in self.children:
            for (childnode, childlevel) in child.walk_with_depth(level+1):
                yield (childnode, childlevel)



[docs]class Form(object):
    """
    Parameters:
        values: An iterable containing a single key-level
            representation to create an unambiguous form, or multiple key-level
            representations to create an ambiguous one.
        levels_to_show: An ordered list

    """
    def __init__(self,
                 levels_to_show: List[Level],
                 values: Iterable[str]) -> None:
        self.levels_to_show = levels_to_show
        self.values = set(values)

    def __invert__(self) -> 'Form':
        return Form(levels_to_show=[], values=self.values)

    def __and__(self, other) -> 'Form':
        return Form(levels_to_show=(self.levels_to_show + other.levels_to_show),
                    values=(self.values & other.values))

    def __rshift__(self, other: Level) -> 'Form':
        return Form(levels_to_show=self.levels_to_show + [other],
                    values=self.values)

    def __repr__(self) -> str:
        readings = self.at_levels(self.levels_to_show)
        self._issue_ambiguity_warnings(readings)
        out = []
        for reading in readings:
            for representation in reading:
                out.append(str(representation))
        return ", ".join(out)

    def __eq__(self, other) -> bool:
        return self.values == other.values

    def _issue_ambiguity_warnings(self, renderings) -> None:
        """ Issue warnings for several kinds of ambiguity. """
        if len(renderings) > 1:
            warn("Rendering a form that is ambiguous at the requested level",
                 SurfaceAmbiguityWarning)
        if len(self.values) > 1:
            warn("Rendering a form that is ambiguous at the root level, but"
                 "not at the requested level",
                 UnderlyingAmbiguityWarning)


[docs]    def is_ambiguous(self) -> bool:
        """ True if this Form is ambiguous --- i.e., if it is consistent with
        multiple underlying root representations. """
        return len(self.values) > 1
